53  std::cout<<"calc time map\n";

58

62

64  for(int i = 0; i < (*m_points).size(); i++){
65  //calc distance
66  float pnt_dis = -1;
67  //std::cout<<(*m_points)[i][0]<<", "<<(*m_points)[i][1]<<"\n";
68  if( m_mode_d == 0 /*abd 1D*/) {
69  pnt_dis = abs(col - (*m_points)[i][0] * m_base->get_width()) + abs(row - (*m_points)[i][1] * m_base->get_height());
70  }
71  else /*euklid 2D*/ {
72  pnt_dis = sqrt(pow(col - (*m_points)[i][0] * m_base->get_width(), 2) + pow(row - (*m_points)[i][1] * m_base->get_height(), 2));
73  }
74  //insert according to distance
75  if(pnts_sorted.size() == 0){
76  pnts_sorted.push_back(cv::Vec3f((*m_points)[i][2], (*m_points)[i][3], pnt_dis));
77  }
78  else{
79  bool inserted = false;
80  for(std::list<cv::Vec3f>::iterator o = pnts_sorted.begin(); o != pnts_sorted.end(); ++o){
81  if((*o)[2] > pnt_dis){
82  pnts_sorted.insert(o, cv::Vec3f((*m_points)[i][2], (*m_points)[i][3], pnt_dis));
83  inserted = true;
84  break;
85  }
86  }
87  if(!inserted){
88  pnts_sorted.push_back(cv::Vec3f((*m_points)[i][2], (*m_points)[i][3], pnt_dis));
89  }
90  }
91  }
92  //calc borders:
93  if(m_num_pnts!=-1){
94  pnts_sorted.resize(m_num_pnts);
95  }
96 
97  float ref_dis = (*pnts_sorted.begin())[2] + (pnts_sorted.back())[2];
98  float start_border = 0;
99  float end_border = 0;
100  float fac_sum = 0;
101 
102  for(std::list<cv::Vec3f>::iterator o = pnts_sorted.begin(); o != pnts_sorted.end(); ++o){
103 
104  if( m_param < 0 /*abs 1D*/){
105  fac_sum += utils::sigmoid(ref_dis - (*o)[2], -m_param, ref_dis / 2);
106  }
107  else /*euklid 2D*/{
108  //fac_sum += sqrt(std::pow(abs(ref_dis - (*o)[2]), m_param));
109  fac_sum += std::pow(abs(1 -((*o)[2]/ref_dis)), m_param);
110  }
111  //std::cout<<std::pow(abs(ref_dis-(*o)[2]),m_param)<<"val\n";
112  }
113 
114  float seg_delta = seg_end - seg_start;
115 
116  for(std::list<cv::Vec3f>::iterator o = pnts_sorted.begin(); o != pnts_sorted.end(); ++o){
117 
118  double influence = 0;
119  if( m_param < 0 /*abs 1D*/){
120  influence = utils::sigmoid(ref_dis - (*o)[2], -m_param, ref_dis / 2) / fac_sum;
121  }
122  else if(m_param > 0){
123  //influence = sqrt(std::pow(abs(ref_dis - (*o)[2]), m_param)) / fac_sum;
124  influence = std::pow(abs(1 -((*o)[2]/ref_dis)), m_param) / fac_sum;
125  }
126 
127  //std::cout<<"influ "<< influence<<"\n";
128  start_border += (*o)[0] * influence;
129  end_border += (*o)[1] * influence;
130  }
131  /*: to map fade [0-1] to segment
132  start_border = seg_start + start_border * seg_delta;
133  end_border = seg_start + end_border * seg_delta;
134  */
135  uc_pixel_map[0] = start_border;
136  uc_pixel_map[1] = end_border;
137  //shift ptr:
138  ptr_map += 2;
139  }
140  }
141 
142  if(error_occured){
143  std::cout<<"was sth. wrong creating time map!\n";
144  }
145  m_time_map[id] = pixel_times;

188 void Timefadepoints::compute_frame(cv::Mat& result, cv::Mat& fac_mat, cv::Mat& current_frame, int frame_num, int seg_id, int sign) {

204  float end_border = uc_pixel_map[1];
205 
206  if(frame_num > (int)start_border && frame_num <= (int)end_border){
207  uc_pixel_res[0] += sign*uc_pixel_current[0];
208  uc_pixel_res[1] += sign*uc_pixel_current[1];
209  uc_pixel_res[2] += sign*uc_pixel_current[2];

216  weight*=sign;

218  uc_pixel_res[1] += weight * uc_pixel_current[1];
219  uc_pixel_res[2] += weight * uc_pixel_current[2];
220  uc_pixel_fac[0] += weight;
221  uc_pixel_fac[1] += weight;
222  uc_pixel_fac[2] += weight;

224  else if(frame_num == (int)end_border + 1){
